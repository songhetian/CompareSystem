## 💡 重构指南

### 使用新组件

#### PageHeader 使用示例

```tsx
import { PageHeader } from '../components/common';

<PageHeader
  title='页面标题'
  subtitle='页面描述'
  icon='🎨'
  extra={
    <Button type='primary'>操作按钮</Button>
  }
/>
```

#### StatsCard 使用示例

```tsx
import { StatsCard } from '../components/common';

<StatsCard
  title='总销售额'
  value={12580}
  suffix='万'
  precision={2}
  trend='up'
  trendValue='+15.3%'
  color='#165DFF'
/>
```

### 使用设计令牌

```tsx
// 间距
style={{ padding: 'var(--spacing-large)' }}

// 圆角
style={{ borderRadius: 'var(--radius-large)' }}

// 阴影
style={{ boxShadow: 'var(--shadow-medium)' }}

// 颜色
style={{ color: 'var(--primary-color)' }}
```

### 标准页面结构

```tsx
export const YourPage = () => {
  return (
    <div className='page-container'>
      {/* 页面头部 */}
      <PageHeader
        title='页面标题'
        subtitle='页面描述'
      />

      {/* 页面内容 */}
      <Card bordered={false}>
        {/* 具体内容 */}
      </Card>
    </div>
  );
};
```
